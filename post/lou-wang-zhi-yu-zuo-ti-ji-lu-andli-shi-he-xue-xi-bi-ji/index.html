经典题，先考虑只有一次查询怎么做。

记 nxtinxt_inxti​ 为下一次出现 aia_iai​ 的位置

首先预处理出 l=1l=1l=1 时的答案。

然后考虑左端点 l→l+1l\rightarrow l+1l→l+1，此时我们要删除 ala_lal​，相当于把 [l+1,nxtl−1][l+1,nxt_l-1][l+1,nxtl​−1] 这些数的答案与 ala_lal​ 取 min⁡\minmin。

但是区间取 min⁡\minmin 很难做，发现 mex\text{mex}mex 满足单调不降，于是可以线段树二分出第一个 >min⁡>\min>min 的位置，然后区间赋值。

考虑拓展到多次询问，我们做个差分，用左端点在 [1,r][1,r][1,r] 的答案减去 [1,l−1][1,l-1][1,l−1] 的答案，然后就变成查询历史和，也就是 ∑l≤k∑r∈[x,y]mex(al,al+1,…,ar)\sum_{l\le k}\sum_{r\in[x,y]} \text{mex}(a_l,a_{l+1},\dots,a_r)∑l≤k​∑r∈[x,y]​mex(al​,al+1​,…,ar​)

发现如果没有修改，历史和一定成一个一次函数的样子，但如果进行了修改，会变成一个分段函数的样子。

分段函数不好维护，但是我们发现将询问离线后，lll 从左到右扫的时候，只有最后一段才是有用的，所以我们只用维护最后一段即可。

由于一次函数具有可加性，我们可以让 k=alk=a_lk=al​，然后把之前的东西加到 bbb 上面。

我们在区间赋值的时候，如果 mex\text{mex}mex 的最大值等于最小值就整体赋值，否则递归下去。可以证明复杂度是对的。因为赋值后整个区间的 mex\text{mex}mex 都变成一样了，所以一个区间最多只会往下递归一次。这样复杂度为 O(nlog⁡n)O(n\log n)O(nlogn)。

#include<bits/stdc++.h>
#define int long long
#define pb push_back
using namespace std;
const int N=1e6+30;
struct que{
	int l,r,id,op;
}; 
struct T{
	int mx,mn,smk,smb,tgk,tgb;
}tr[N*4];
int typ,n,Q,a[N],nxt[N],l,r,p[N],ans[N],mex;
bool vis[N];
vector<que> q[N];
inline int read(){
    int x=0,f=1;
    char ch=getchar();
    while(ch<'0'||ch>'9'){if(ch=='-')f=-1;ch=getchar();}
    while(ch>='0'&&ch<='9'){x=(x<<1)+(x<<3)+(ch^48);ch=getchar();}
    return x*f;
}
inline void write(int x){
    if(x<0){x=~(x-1);putchar('-');}
    if(x>9)write(x/10);
    putchar(x%10+'0');
}
void mg(int i){
	tr[i].mn=min(tr[i<<1].mn,tr[i<<1|1].mn);
	tr[i].mx=max(tr[i<<1].mx,tr[i<<1|1].mx);
	tr[i].smk=tr[i<<1].smk+tr[i<<1|1].smk;
	tr[i].smb=tr[i<<1].smb+tr[i<<1|1].smb;
}
void addtg(int i,int l,int r,int k,int b){//cout<<l<<" "<<r<<"\n";
	tr[i].tgk+=k,tr[i].tgb+=b;
	tr[i].smk+=(r-l+1)*k,tr[i].smb+=(r-l+1)*b;
	tr[i].mn+=k,tr[i].mx+=k;
}
void build(int i,int l,int r){
	if(l==r){
		vis[a[l]]=1;
		while(vis[mex]) mex++;//cout<<i<<' '<<l<<' '<<mex<<"\n";
		tr[i].mx=tr[i].mn=tr[i].smk=mex;
		return;
	}
	int mid=(l+r)>>1;
	build(i<<1,l,mid);
	build(i<<1|1,mid+1,r);
	mg(i);
}
void pushdown(int i,int l,int r){
	if(tr[i].tgk||tr[i].tgb){
		int mid=(l+r)>>1;
		addtg(i<<1,l,mid,tr[i].tgk,tr[i].tgb);
		addtg(i<<1|1,mid+1,r,tr[i].tgk,tr[i].tgb);
		tr[i].tgk=tr[i].tgb=0;
	}
}
void upd(int i,int l,int r,int x,int y,int k,int b){
	if(tr[i].mx<=b) return;
	if(l>=x&&r<=y&&tr[i].mx==tr[i].mn){//cout<<i<<' '<<l<<' '<<r<<' '<<k<<' '<<b<<' '<<k*(tr[i].mx-b)<<'\n';
		addtg(i,l,r,b-tr[i].mx,k*(tr[i].mx-b));
		return;
	}
	pushdown(i,l,r);
	int mid=(l+r)>>1;
	if(x<=mid) upd(i<<1,l,mid,x,y,k,b);
	if(y>mid) upd(i<<1|1,mid+1,r,x,y,k,b);
	mg(i);
}
int qry(int i,int l,int r,int x,int y,int k){
	// cout<<i<<" "<<l<<" "<<r<<' '<<tr[i].smk<<' '<<tr[i].smb<<"\n";
	if(l>=x&&r<=y) return k*tr[i].smk+tr[i].smb;
	pushdown(i,l,r);
	int mid=(l+r)>>1,res=0;
	if(x<=mid) res+=qry(i<<1,l,mid,x,y,k);
	if(y>mid) res+=qry(i<<1|1,mid+1,r,x,y,k);
	return res;
}
signed main(){
	typ=read(),n=read();
	for(int i=1;i<=n;i++) a[i]=min(read(),n+1);
	for(int i=0;i<=n+1;i++) p[i]=n+1;
	for(int i=n;i;i--){
		nxt[i]=p[a[i]];
		p[a[i]]=i;
	}
	Q=read();
	for(int i=1;i<=Q;i++){
		l=read(),r=read();
		q[r].pb((que){l,r,i,1});
		q[l-1].pb((que){l,r,i,-1});
	}
	build(1,1,n);
	for(int i=1;i<=n;i++){
		for(auto x:q[i])
			ans[x.id]+=x.op*qry(1,1,n,x.l,x.r,i);
		if(i+1<=nxt[i]-1) upd(1,1,n,i+1,nxt[i]-1,i,a[i]);//cout<<i<<" "<<nxt[i]<<"\n";
		upd(1,1,n,i,i,i,0);
	}
	for(int i=1;i<=Q;i++) write(ans[i]),putchar('\n');
}
